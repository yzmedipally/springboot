Pipeline {
    
	node any {
	
		git 'https://github.com/yzmedipally/springboot.git'

		def project_path = "spring-boot-samples/spring-boot-sample-atmosphere/Jenkinsfile.txt"
		
		dir(project_path) {
				stage ("clean"){
					  bat 'mvn clean'
				}
				stage ("verify"){
					  bat 'mvn verify'
				} 
				stage ("compile"){
					  bat 'mvn compile'
				}
				stage ("package"){
					  bat 'mvn package'
				}
				stage ("Install"){
					  bat 'mvn install'
				}
				stage ("Artifact"){
					  archiveArtifacts "target/*.jar"
				}
		}
	}

}
